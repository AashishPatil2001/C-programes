// sum of array 

#include <stdio.h>

int main() {
    int a[10], i, sum=0;
    printf("Enter a 10 value \n");
    for(i=0; i<10; i++){
        printf("Enter a number at %d position : ", i+1);
        scanf("%d", &a[i]);
    }
    for(i=0; i<10; i++){
        sum=sum+a[i];
    }
    printf("Sum of array is %d", sum);
    return 0;
}
